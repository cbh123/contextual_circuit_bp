DROP TABLE IF EXISTS experiments
DROP TABLE IF EXISTS performance
DROP TABLE IF EXISTS in_process

CREATE TABLE experiments (_id bigserial primary key, experiment_name varchar, model_struct varchar, loss_function varchar, regularization_type varchar, regularization_strength float, optimizer varchar, lr float, dataset varchar)

CREATE TABLE performance (_id bigserial primary key, experiment_name varchar, experiment_id int, summary_dir varchar, ckpt_file varchar, time_elapsed float, training_step int, validation_loss float, training_loss float)

CREATE TABLE in_process (experiment_id int, experiment_name varchar)

ALTER TABLE  experiments ADD CONSTRAINT unique_exp UNIQUE (experiment_name, model_struct, loss_function, regularization_type, regularization_strength, optimizer, lr, dataset)
